    def initialize
      # rss tags
        @rss_tags = [
          {
                  :image     => nil,
          :textInput => nil,
          :skipHours => nil,
          :skipDays  => nil,
          :items     => [{ 'rdf:Seq''rdf:Seq' => nil }],
         #:item      => @rss_item_tags
                }
        ]
      @rss_item_tags = [ {} ]

      #atom tags
      @atom_tags = [
        {
          :author       => nil,
          :contributor  => nil,
         #:entry        => @atom_entry_tags
        }
      ]
      @atom_entry_tags = [
        {
          :author       => nil, 
          :contributor  => nil
        }
      ]
    
      # tags whose value is a date
      @date_tags = [ :pubDate, :lastBuildDate, :published, :updated, 'dc:date''dc:date', :expirationDate ]
  
      # tags that can be used as the default value for a tag with attributes
      @value_tags = [ CONTENT_KEY, :href ]
  
      # tags that don't become part of the parsed object tree
      @ghost_tags = [ 'rdf:Seq''rdf:Seq' ]
    
      # tag/attribute aliases
        @aliases = {
          :items        => :item_array,
          :item_array   => :entry_array,
          :entries      => :entry_array,
          :entry_array  => :item_array,
          :link         => 'link+self''link+self'
        }
        
        # bang mods
        @bang_mods = [ :stripHtml ]
        @bang_mod_fns = {
          :stripHtml => proc {|str| str.gsub(/<\/?[^>]*>/, "").strip },
          :wrap      => proc {|str, col| str.gsub(/(.{1,#{col}})( +|$\n?)|(.{1,#{col}})/, "\\1\\3\n").strip }
        }
    end
